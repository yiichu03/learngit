import os
import cv2
import numpy as np
from skimage.feature import hog
from sklearn.decomposition import PCA
from sklearn.svm import SVC

from sklearn.model_selection import train_test_split

# å‡è®¾ä½ å·²ç»ä¸‹è½½å¹¶è§£å‹äº†VOC2007æ•°æ®é›†åˆ°'./VOC2007/'

# é€‰æ‹©"person"ç±»åˆ«çš„annotationåˆ—è¡¨æ–‡ä»¶
# æ­¤æ–‡ä»¶ä¸­æ¯è¡Œæ ¼å¼ä¸ºï¼š<image_id> <label(1æˆ–-1)>
# å¦‚ï¼š'ImageSets/Main/person_trainval.txt'

    lines = f.readlines()

    parts = line.strip().split()
    image_id = parts[0]
    label = int(parts[1])
    image_ids.append(image_id)
    labels.append(label)

    if not os.path.exists(img_path):
        continue

    if img is None:
        continue
    
    # ç®€åŒ–å¤„ç†ï¼šä½¿ç”¨æ•´å¼ å›¾åƒç¼©æ”¾å¹¶è½¬ç°åº¦æå–HOGï¼ˆå¦‚éœ€è¦å¯æ ¹æ®personçš„boxåšè£å‰ªï¼‰
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # æå–HOGç‰¹å¾
    hog_feat = hog(gray, 
                   orientations=9, 
                   pixels_per_cell=(8, 8), 
                   cells_per_block=(2, 2), 
                   block_norm='L2-Hys', 
                   transform_sqrt=True, 
                   feature_vector=True)

pca = PCA(n_components=50, whiten=True, random_state=42)

X_test_pca = pca.transform(X_test)

# ä½¿ç”¨SVMåˆ†ç±»

y_pred = svm_clf.predict(X_test_pca)

print("Classification Report:")

print("Confusion Matrix:")

The PASCAL VOC 2007 dataset contains 9,963 images labeled with 20 object categories. We used the "person" and "non-person" binary classification task.
